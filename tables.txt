export type GameEventMap = {
// Combat
MONSTER_KILLED:    { monsterId: string; zoneId: string; loot: ItemDrop[]; xpGained: number }
PLAYER_DAMAGED:    { damage: number; source: string; newHp: number }
PLAYER_DIED:       { zoneId: string; killedBy: string }
ATTACK_DODGED:     { attackerId: string }
COMBAT_STARTED:    { zoneId: string; monsterId: string }
COMBAT_ENDED:      { zoneId: string; victory: boolean }

// Métiers
SKILL_ACTION_DONE: { skillId: SkillId; actionId: string; xpGained: number; outputs: ItemDrop[] }
SKILL_LEVEL_UP:    { skillId: SkillId; newLevel: number }
SKILL_MILESTONE:   { skillId: SkillId; level: number; bonus: string }

// Craft
CRAFT_COMPLETED:   { recipeId: string; skillId: SkillId; output: ItemDrop; xpGained: number }
CRAFT_FAILED:      { recipeId: string; reason: string }
UPGRADE_BUILT:     { upgradeId: string; level: number }

// Inventaire
ITEM_ADDED:        { itemId: string; quantity: number; source: string }
ITEM_REMOVED:      { itemId: string; quantity: number }
ITEM_EQUIPPED:     { itemId: string; slot: SlotId }
INVENTORY_FULL:    { itemId: string; quantityLost: number }

// Agriculture
PLANT_SOWED:       { plotIndex: number; seedId: string; readyAt: number }
PLANT_READY:       { plotIndex: number; seedId: string }
PLANT_HARVESTED:   { plotIndex: number; seedId: string; outputs: ItemDrop[]; perfect: boolean }

// Économie
ITEM_SOLD:         { itemId: string; quantity: number; goldGained: number }
ITEM_BOUGHT:       { itemId: string; quantity: number; goldSpent: number }
GOLD_CHANGED:      { delta: number; newTotal: number; source: string }

// Quêtes
QUEST_STARTED:     { questId: string }
QUEST_OBJECTIVE:   { questId: string; objectiveIndex: number; current: number; target: number }
QUEST_COMPLETED:   { questId: string; rewards: QuestReward }
DAILY_RESET:       { date: string; newQuestIds: string[] }

// Session
GAME_LOADED:       { isGuest: boolean; offlineDuration: number }
OFFLINE_RECOVERED: { duration: number; summary: OfflineSummary }
SAVE_COMPLETED:    { target: 'local' | 'cloud' }
}

type Listener<T> = (payload: T) => void

class GameEventBus {
private listeners = new Map<string, Set<Listener<any>>>()

on<K extends keyof GameEventMap>(event: K, listener: Listener<GameEventMap[K]>): () => void {
if (!this.listeners.has(event)) this.listeners.set(event, new Set())
this.listeners.get(event)!.add(listener)
return () => this.listeners.get(event)?.delete(listener)  // unsubscribe fn
}

emit<K extends keyof GameEventMap>(event: K, payload: GameEventMap[K]): void {
this.listeners.get(event)?.forEach(fn => {
try { fn(payload) }
catch (err) { console.error(`[EventBus] Error in ${event} listener:`, err) }
})
}

// Vider tous les listeners (utile pour les tests)
clear(): void { this.listeners.clear() }
}

export const bus = new GameEventBus()  // singleton partagé

// Validation au démarrage (appelée dans app/layout.tsx)
export function validateGameData(): void {
const errors: string[] = []
// Vérifier que chaque output de recette pointe vers un item existant
rawRecipes.recipes.forEach(r => {
r.inputs?.forEach(i => { if (!itemMap.has(i.itemId)) errors.push(`Recipe ${r.id}: unknown input ${i.itemId}`) })
if (r.output && !itemMap.has(r.output.itemId)) errors.push(`Recipe ${r.id}: unknown output ${r.output.itemId}`)
})
// Vérifier que chaque drop de monstre pointe vers un item existant
rawZones.monsters.forEach(m => {
m.drops?.forEach(d => { if (!itemMap.has(d.itemId)) errors.push(`Monster ${m.id}: unknown drop ${d.itemId}`) })
})
if (errors.length > 0) throw new Error(`Data validation failed:\n${errors.join('\n')}`)
}

// NIVEAU 1 : Validation de l'entrée
function validateDelta(rawDelta: number): number {
if (!Number.isFinite(rawDelta) || rawDelta < 0) {
console.error('[Offline] Invalid delta:', rawDelta)
return 0
}
const cap = FLAGS.ENABLE_PRAYER_SKILL  // exemple flag qui change le cap
? OFFLINE_CAP_GRIMOIRE_H * 3600 * 1000
: OFFLINE_CAP_DEFAULT_H  * 3600 * 1000
return Math.min(rawDelta, cap)
}

// NIVEAU 2 : Validation de chaque résultat intermédiaire
function safeXp(base: number, count: number): number {
const result = base * count
if (!Number.isFinite(result) || result < 0) return 0
return Math.floor(result)
}

// NIVEAU 3 : Validation de l'inventaire avant application
function applyOfflineLoot(store: GameStore, loot: Record<string, number>): string[] {
const warnings: string[] = []
for (const [itemId, qty] of Object.entries(loot)) {
if (!Number.isFinite(qty) || qty <= 0) { warnings.push(`Invalid qty for ${itemId}`); continue }
const current = store.inventory[itemId] ?? 0
const item    = GameData.item(itemId)
const capped  = Math.min(qty, item.stackMax - current)
if (capped < qty) warnings.push(`Stack overflow: ${itemId} (lost ${qty - capped})`)
if (capped > 0) store.inventory[itemId] = current + capped
}
return warnings  // affichés dans l'écran de récupération
}

type MigrationFn = (state: Record<string, any>) => Record<string, any>

const migrations: Record<number, MigrationFn> = {
// v1 → v2 : ajout du système d'upgrades
1: (s) => ({ ...s, upgrades: {}, unlockedFlags: [] }),

// v2 → v3 : combat décomposé en 8 skills
2: (s) => ({
...s,
skills: {
...s.skills,
// Migrer l'ancien skill 'combat' vers 'attack' + stats dérivées
attack:       { level: s.skills.combat?.level ?? 1, xp: s.skills.combat?.xp ?? 0, activeAction: null, actionStartedAt: null },
strength:     { level: 1, xp: 0, activeAction: null, actionStartedAt: null },
ranged:       { level: 1, xp: 0, activeAction: null, actionStartedAt: null },
magic:        { level: 1, xp: 0, activeAction: null, actionStartedAt: null },
defense:      { level: 1, xp: 0, activeAction: null, actionStartedAt: null },
dodge:        { level: 1, xp: 0, activeAction: null, actionStartedAt: null },
constitution: { level: s.skills.combat?.level ?? 1, xp: 0, activeAction: null, actionStartedAt: null },
prayer:       { level: 1, xp: 0, activeAction: null, actionStartedAt: null },
},
}),
}

export function migrateSave(raw: Record<string, any>): GameState {
let state = structuredClone(raw)  // ne jamais muter l'original
const startVersion = state.version ?? 1

for (let v = startVersion; v < SAVE_VERSION; v++) {
const migrate = migrations[v]
if (!migrate) throw new Error(`No migration from v${v} to v${v + 1}`)
state = { ...migrate(state), version: v + 1 }
console.info(`[Save] Migrated from v${v} to v${v + 1}`)
}
return state as GameState
}

import { bus } from './event-bus'
import { GameData } from './data-loader'

// Map des handlers par type d'objectif
type ObjectiveHandler = (
objective: QuestObjective,
event: any,
current: number
) => number  // retourne la nouvelle valeur de progression

const OBJECTIVE_HANDLERS: Record<string, ObjectiveHandler> = {

kill: (obj, event: GameEventMap['MONSTER_KILLED'], current) => {
if (obj.monsterId && obj.monsterId !== event.monsterId) return current
if (obj.zoneId   && obj.zoneId   !== event.zoneId)   return current
return current + 1
},

gather: (obj, event: GameEventMap['SKILL_ACTION_DONE'], current) => {
if (obj.skillId && obj.skillId !== event.skillId) return current
if (obj.actionId && obj.actionId !== event.actionId) return current
return current + 1
},

craft: (obj, event: GameEventMap['CRAFT_COMPLETED'], current) => {
if (obj.recipeId && obj.recipeId !== event.recipeId) return current
if (obj.skillId  && obj.skillId  !== event.skillId)  return current
return current + event.output.qty
},

harvest: (obj, event: GameEventMap['PLANT_HARVESTED'], current) => current + 1,

sell: (obj, event: GameEventMap['ITEM_SOLD'], current) => {
if (obj.itemId && obj.itemId !== event.itemId) return current
return current + event.goldGained
},

deliver: (obj, event: GameEventMap['ITEM_REMOVED'], current) => {
if (event.itemId !== obj.itemId) return current
return current + event.quantity
},

reach_level: (obj, event: GameEventMap['SKILL_LEVEL_UP'], current) => {
if (event.skillId !== obj.skillId) return current
return event.newLevel  // la cible est un niveau, pas un compteur
},

kill_boss: (obj, event: GameEventMap['MONSTER_KILLED'], current) => {
const monster = GameData.monster(event.monsterId)
if (monster.rarity !== 'boss') return current
if (obj.zoneId && obj.zoneId !== event.zoneId) return current
return current + 1
},
}

engine/quest-engine.ts — Initialisation et listeners
export function initQuestEngine(store: QuestStore): void {

// Listener générique — routing vers le bon handler
function handleEvent(eventType: string, payload: any): void {
const activeQuests = store.getActiveQuests()

for (const quest of activeQuests) {
let questUpdated = false

quest.objectives.forEach((obj, idx) => {
if (obj.type !== eventType) return
if (obj.completed) return

const handler = OBJECTIVE_HANDLERS[obj.type]
if (!handler) return

const current  = quest.progress[idx] ?? 0
const newValue = handler(obj, payload, current)

if (newValue !== current) {
store.updateObjectiveProgress(quest.id, idx, Math.min(newValue, obj.qty))
bus.emit('QUEST_OBJECTIVE', { questId: quest.id, objectiveIndex: idx,
current: Math.min(newValue, obj.qty), target: obj.qty })
questUpdated = true
}
})

// Vérifier si la quête est complète
if (questUpdated && store.isQuestComplete(quest.id)) {
completeQuest(quest.id, store)
}
}
}

// Brancher les événements pertinents sur le handler générique
bus.on('MONSTER_KILLED',    p => { handleEvent('kill', p); handleEvent('kill_boss', p) })
bus.on('SKILL_ACTION_DONE', p => handleEvent('gather', p))
bus.on('CRAFT_COMPLETED',   p => handleEvent('craft', p))
bus.on('PLANT_HARVESTED',   p => handleEvent('harvest', p))
bus.on('ITEM_SOLD',         p => handleEvent('sell', p))
bus.on('ITEM_REMOVED',      p => handleEvent('deliver', p))
bus.on('SKILL_LEVEL_UP',    p => handleEvent('reach_level', p))
}

function completeQuest(questId: string, store: QuestStore): void {
const quest   = GameData.quest(questId)
store.markCompleted(questId)
// Créditer les récompenses
applyQuestRewards(quest.rewards, store)
// Débloquer les quêtes suivantes
quest.unlocks?.forEach(id => store.unlockQuest(id))
bus.emit('QUEST_COMPLETED', { questId, rewards: quest.rewards })
}

import { mulberry32 } from './rng'

// Seed unique par joueur par jour
function getDailySeed(playerId: string): number {
const dayIndex = Math.floor(Date.now() / 86_400_000)  // UTC
return dayIndex ^ hashString(playerId)
}

function getWeeklySeed(playerId: string): number {
const weekIndex = Math.floor(Date.now() / 604_800_000)
return weekIndex ^ hashString(playerId)
}

// Hash FNV-1a simple (déterministe, pas crypto)
function hashString(str: string): number {
let hash = 2166136261
for (const c of str) hash = Math.imul(hash ^ c.charCodeAt(0), 16777619)
return hash >>> 0  // unsigned 32-bit
}

export function generateDailyQuests(
playerId: string,
playerSkills: Record<SkillId, { level: number }>,
count = 3
): string[] {
const rng  = mulberry32(getDailySeed(playerId))
const pool = rawQuests.dailyQuestPool.filter(q =>
isQuestAccessible(q, playerSkills)
)
// Sélection sans remise
const selected: string[] = []
const available = [...pool]
while (selected.length < count && available.length > 0) {
const idx = Math.floor(rng() * available.length)
selected.push(available[idx].id)
available.splice(idx, 1)
}
return selected
}

export function generateWeeklyQuest(playerId: string): string {
const rng = mulberry32(getWeeklySeed(playerId))
const idx = Math.floor(rng() * rawQuests.weeklyQuests.length)
return rawQuests.weeklyQuests[idx].id
}

// Vérifier si une quête est accessible selon les niveaux du joueur
function isQuestAccessible(quest: any, skills: Record<string, { level: number }>): boolean {
if (!quest.reqSkills) return true
return Object.entries(quest.reqSkills).every(
([skillId, minLevel]) => (skills[skillId]?.level ?? 1) >= (minLevel as number)
)
}

export function checkDailyReset(store: QuestStore, playerId: string): void {
const todayStr = new Date().toISOString().slice(0, 10)  // 'YYYY-MM-DD'

if (store.lastDailyDate === todayStr) return  // déjà resetté aujourd'hui

const newQuestIds = generateDailyQuests(playerId, store.playerSkills)
store.setDailyQuests(newQuestIds, todayStr)

// Reset du streak si un jour a été manqué
const yesterday = new Date(Date.now() - 86_400_000).toISOString().slice(0, 10)
if (store.lastDailyDate !== yesterday) {
store.resetStreak()
} else {
store.incrementStreak()
}

bus.emit('DAILY_RESET', { date: todayStr, newQuestIds })
}

import { calculateOfflineProgress } from '../engine/offline-engine'
import { makeDefaultGameState }      from '../engine/test-helpers'
import { GameData }                  from '../engine/data-loader'
import { CONSTANTS }                 from '../engine/constants'

const SIMULATION_HOURS = 100
const TICK_SIZE_MS     = 60_000  // ticks d'1 minute pour la simulation

async function runBalanceSimulation() {
const store = makeDefaultGameState()
const report: BalanceReport = { xp: {}, economy: {}, combat: {}, craft: {} }

// Simuler heure par heure
for (let h = 0; h < SIMULATION_HOURS; h++) {
const deltaMs = 3600_000
const result  = calculateOfflineProgress(store, store.lastSaveAt + deltaMs)
applyOfflineResult(store, result)
store.lastSaveAt += deltaMs

// Snapshot toutes les 10h
if (h % 10 === 0) snapshotMetrics(report, store, h)
}

printReport(report)
}

function printReport(report: BalanceReport): void {
console.log('\n=== IDLE REALMS — BALANCE REPORT ===')

console.log('\n📈 XP PROGRESSION (heures pour atteindre le niveau)')
for (const [skillId, timeline] of Object.entries(report.xp)) {
console.log(`  ${skillId.padEnd(15)} | niv.10: ${timeline[10]?.toFixed(1)}h | niv.50: ${timeline[50]?.toFixed(1)}h | niv.99: ${timeline[99]?.toFixed(1)}h`)
}

console.log('\n💰 ÉCONOMIE (par heure)')
for (const [source, gold] of Object.entries(report.economy)) {
console.log(`  ${source.padEnd(20)} : ${gold.toFixed(0)} or/h`)
}

console.log('\n⚔️  COMBAT')
for (const [zone, stats] of Object.entries(report.combat)) {
console.log(`  ${zone}: survie ${(stats.survivalRate * 100).toFixed(0)}% | potions: ${stats.potionsPerHour.toFixed(1)}/h`)
}
}
